649 void Drv_Devices_loadChannelSettings(uint8_t parameter)
650 {
651  if(channelSettingsCounter == 0)// if it's the first byte in this channel's array, this byte is the channel number to set
652  {
653  currentChannelToSet = Drv_Devices_getChannelNumber(parameter); // we just got the channel to load settings into (shift number down for array usage)
654  channelSettingsCounter++;
655 
656  if(!is_running)
657  {
658  uint8_t number = currentChannelToSet + 1 + '0';
659  Drv_UART_Send((uint8_t*)("Load settings of channel: "),26);
660  Drv_UART_Send(&number,1);
661  Drv_UART_Send((uint8_t*)("\r\n"),2);
662  }
663 
664  }
665  else
666  {
667 
668  // setting bytes are in order: POWER_DOWN, GAIN_SET, INPUT_TYPE_SET, BIAS_SET, SRB2_SET, SRB1_SET
669 
670  parameter = parameter - '0'; //convert char to number
671 
672  if(channelSettingsCounter-1 == GAIN_SET)
673  {
674  parameter <<= 4;
675  }
676 
677  // Fill array with current channel settings defined in the Drv_ADS1299 library
678  channelSettings[currentChannelToSet][channelSettingsCounter-1] = parameter;
679 
680  channelSettingsCounter++;
681 
682  if(channelSettingsCounter == 7) // 1 currentChannelToSet, plus 6 channelSetting parameters
683  {
684  getChannelSettings = FALSE;
685  }
686  }
